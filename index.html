  <script>

    // Скролл к секции

    function scrollToSection(id) {

      document.getElementById(id).scrollIntoView({ behavior: 'smooth' });

      document.querySelectorAll('nav button').forEach(btn => btn.classList.remove('active'));

      event.target.classList.add('active');

    }


    // Калькулятор

    function calculate() {

      const l = parseFloat(document.getElementById('length').value) || 0;

      const w = parseFloat(document.getElementById('width').value) || 0;

      const h = parseFloat(document.getElementById('height').value) || 0;

      const weight = parseFloat(document.getElementById('weight').value) || 0;

      

      if (l <= 0 || w <= 0 || h <= 0 || weight < 30) {

        alert('Заполните все поля корректно! Вес минимум 30 кг.');

        return;

      }

      

      const volume = (l * w * h) / 1000000; // м³

      if (volume === 0) return;

      const density = weight / volume;

      

      let pricePerKg;

      if (density > 400) pricePerKg = 3.00;

      else if (density >= 350) pricePerKg = 3.10;

      else if (density >= 300) pricePerKg = 3.20;

      else if (density >= 250) pricePerKg = 3.30;

      else if (density >= 200) pricePerKg = 3.40;

      else if (density >= 190) pricePerKg = 3.50;

      else if (density >= 180) pricePerKg = 3.60;

      else if (density >= 170) pricePerKg = 3.70;

      else if (density >= 160) pricePerKg = 3.80;

      else if (density >= 150) pricePerKg = 3.90;

      else if (density >= 140) pricePerKg = 4.00;

      else if (density >= 130) pricePerKg = 4.10;

      else if (density >= 120) pricePerKg = 4.20;

      else if (density >= 110) pricePerKg = 4.30;

      else pricePerKg = 4.40; // <100 или <110

      

      const total = pricePerKg * weight;

      

      const result = document.getElementById('result');

      result.style.display = 'block';

      result.innerHTML = `

        <strong>Коэффициент плотности:</strong> ${density.toFixed(2)} кг/м³<br>

        <strong>Цена за 1 кг:</strong> $${pricePerKg.toFixed(2)}<br>

        <strong style="font-size:1.2rem;">Общая стоимость за груз (${weight} кг):</strong> $${total.toFixed(2)}

      `;


      // Показать секцию "Заказать доставку" с контактами

      document.getElementById('orderSection').style.display = 'block';

    }
